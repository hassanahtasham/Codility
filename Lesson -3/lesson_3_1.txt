class Solution {
    public int solution(int X, int Y, int D) {

        int steps = 0;
        int diff = Y-X;
        if(diff == 0){
            return 0;
        }

        if(diff % D == 0){
            steps = diff/D;
        }else{
            steps = diff/D + 1;

        }
      
      return steps;

    }
}
